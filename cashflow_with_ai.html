גיבוי Google Drive

☁️ הגדר גיבוי אוטומטי ל-Google Drive

הגיבוי ישמור את כל הנתונים שלך בקובץ JSON בדרייב, פעם ביום אוטומטית.

  1. עבור אל script.google.com וצור פרויקט חדש (New Project)
  2. מחק את הקוד הקיים והדבק את הקוד הבא: 
    function doPost(e) {
  const data = JSON.parse(e.postData.contents);
  const token = data.token;
  if (token !== PropertiesService.getScriptProperties().getProperty('SECRET_TOKEN')) {
    return ContentService.createTextOutput(JSON.stringify({error:'Unauthorized'}))
      .setMimeType(ContentService.MimeType.JSON);
  }
  const folder = DriveApp.getRootFolder();
  const fileName = 'cashflow-backup-' + new Date().toISOString().slice(0,10) + '.json';
  const existing = folder.getFilesByName('cashflow-backup-latest.json');
  if (existing.hasNext()) existing.next().setTrashed(true);
  folder.createFile('cashflow-backup-latest.json', data.payload, MimeType.PLAIN_TEXT);
  folder.createFile(fileName, data.payload, MimeType.PLAIN_TEXT);
  return ContentService.createTextOutput(JSON.stringify({ok:true, file: fileName}))
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  const token = e.parameter.token;
  if (token !== PropertiesService.getScriptProperties().getProperty('SECRET_TOKEN')) {
    return ContentService.createTextOutput(JSON.stringify({error:'Unauthorized'}))
      .setMimeType(ContentService.MimeType.JSON);
  }
  const folder = DriveApp.getRootFolder();
  const files = folder.getFilesByName('cashflow-backup-latest.json');
  if (!files.hasNext()) {
    return ContentService.createTextOutput(JSON.stringify({ok:false, error:'No backup found'}))
      .setMimeType(ContentService.MimeType.JSON);
  }
  const content = files.next().getBlob().getDataAsString();
  return ContentService.createTextOutput(JSON.stringify({ok:true, payload: content}))
    .setMimeType(ContentService.MimeType.JSON);
}
  3. לחץ על Project Settings (גלגל שיניים) → Script Properties → הוסף: SECRET_TOKEN עם ערך סיסמה כלשהי שתבחר
  4. לחץ Deploy → New Deployment → סוג: Web App → "Execute as: Me" → "Who has access: Anyone" → לחץ Deploy → העתק את ה-URL
  5. הדבק את ה-URL והסיסמה שלך למטה ולחץ "חבר"
